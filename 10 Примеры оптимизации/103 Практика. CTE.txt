postgres=# ------------------------------- создание CTE запросов при обработке данных:
create table orders as                                                          -- таблица продаж
select (random() * 10)::DECIMAL(12,2) as amount                                 -- заполняем случайными данными
    , (random() * 10)::DECIMAL(12,3) as quantity
    , (array['Msk', 'Spb', 'Others'])[floor(random() * 3 + 1)] as region        -- продажи по трём регионам, случайным образом
from generate_series(1, 50000);


WITH regional_sales AS (                                                        -- подзапрос CTE "regional_sales"
    SELECT region, SUM(amount) AS total_sales                                   -- сумма продаж по региону
    FROM orders                                                                 -- из таблицы "orders"
    GROUP BY region                                                             -- сгруппированные по региону, и
), top_regions AS (                                                             -- подзапрос "top_regions", который будет использовать результат подзапроса "regional_sales"
    SELECT region
    FROM regional_sales                                                         -- использцет результат подзапроса "regional_sales"
    WHERE total_sales > (SELECT SUM(total_sales)/3 FROM regional_sales)         -- где вывод рассчитывает по формуле "больше, чем 1/3 прдаж"
)
SELECT region,
       SUM(quantity) AS product_units,
       SUM(amount) AS product_sales
FROM orders                                                                     -- запрос CTE в таблицу "orders"
WHERE region IN (SELECT region FROM top_regions)                                -- где "region" выбран из ТОР-регионов
GROUP BY region;
SELECT 50000

------------ запросы без CTE:

WITH top_regions AS (
    SELECT a.region
    FROM (
        SELECT region, SUM(amount) AS total_sales
        FROM orders
        GROUP BY region
    ) a
    WHERE total_sales > (SELECT SUM(total_sales)/3 FROM (
            SELECT region, SUM(amount) AS total_sales
            FROM orders
            GROUP BY region
            ) b
        )
)
SELECT region,
       SUM(quantity) AS product_units,
       SUM(amount) AS product_sales
FROM orders
WHERE region IN (SELECT region FROM top_regions)
GROUP BY region;
